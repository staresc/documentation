 11class StarescHandler:
 12    """Generic handler
 13    
 14    This is a generic handler for StarescExporter
 15    """
 16
 17    out: str
 18
 19    def __init__(self, out: str) -> None:
 20        self.out = out
 21
 22    def import_handler(self, o: Output):
 23        """Import Handler
 24        
 25        This method is called on the hadler when a new output is added to the 
 26        queue. Each handler can decide to either do nothing or perform special
 27        operations, such as the StarescStdoutHandler, which prints the 
 28        vulnerabilities when discovered in the Output
 29        """
 30        pass
 31
 32    def export_handler(self, outputs: list[Output], outfile: str):
 33        """Export Handler
 34        
 35        This method is called on the hadler when the execution is over and the
 36        whole queue is converted to a list.
 37        """
 38        pass

 41class StarescCSVHandler(StarescHandler):
 42
 43    def import_handler(self, o: Output):
 44        pass
 45
 46    def export_handler(self, outputs: list[Output], outfile: str):
 47        
 48        def technical_details(o: Output):
 49            tech_details = ""
 50            for i in range(len(o.test_results)):
 51                tech_details += f"cmd: {o.test_results[i]['stdin']}\n"
 52                tech_details += f"stdout: {o.test_results_parsed[i]['stdout']}\n"
 53                tech_details += f"stderr: {o.test_results_parsed[i]['stderr']}\n"
 54                tech_details += "\n\n\n"
 55            return tech_details
 56
 57        def complete_log(o: Output):
 58            ret = []
 59            for test_res in o.test_results:
 60                ret.append({
 61                    "stdin": test_res["stdin"],
 62                    "stdout": test_res["stdout"],
 63                    "stderr": test_res["stderr"],
 64                })
 65            return str(ret)
 66
 67        COLUMNS_TO_FUNC = {
 68            "Host IP"            : lambda x: Connection.get_hostname(x.target.connection),
 69            "Port"               : lambda x: Connection.get_port(x.target.connection),
 70            "Scheme"             : lambda x: Connection.get_scheme(x.target.connection),
 71            "Vulnerable"         : lambda x : x.is_vuln_found(),
 72            "Any timeout"        : lambda x : any(x.get_timeouts()),
 73            "CVSS score"         : lambda x : getattr(x.plugin, 'cvss', 0.0),
 74            "Vulnerability name" : lambda x : getattr(x.plugin, 'name'),
 75            "Description"        : lambda x : getattr(x.plugin, 'description'),
 76            "Technical details"  : technical_details,
 77            "Remediation"        : lambda x : getattr(x.plugin, 'remediation', ''),
 78            "CVE"                : lambda x : getattr(x.plugin, 'CVE', ''),
 79            "CVSS vector"        : lambda x : getattr(x.plugin, 'cvss_vector', ''),
 80            "Complete log"       : complete_log,
 81        }
 82        out_rows = [COLUMNS_TO_FUNC.keys()]
 83        
 84        # it works only in non multithread environment
 85        for output in outputs:
 86            tmp_row = []
 87            for col in COLUMNS_TO_FUNC.keys():
 88                tmp_row.append(COLUMNS_TO_FUNC[col](output))
 89            out_rows.append(tmp_row)
 90
 91        with open(outfile, 'w') as f:
 92            csv_writer = csv.writer(f, delimiter=';')
 93            csv_writer.writerows(out_rows)

 96class StarescStdoutHandler(StarescHandler):
 97
 98    logger: StarescLogger = StarescLogger()
 99    scan_summary: dict
100
101    def __init__(self, out: str) -> None:
102        self.scan_summary = {}
103        super().__init__(out)
104
105    def import_handler(self, o: Output):
106        self.logger.print_if_vuln(o)
107
108        if o.is_vuln_found():
109            host = o.target.get_hostname(o.target.connection)
110            port = o.target.get_port(o.target.connection)
111
112            if f"{host}:{port}" not in self.scan_summary:
113                self.scan_summary[f"{host}:{port}"] = {}
114
115            if o.plugin.severity in self.scan_summary[f"{host}:{port}"]:
116                self.scan_summary[f"{host}:{port}"][o.plugin.severity] += 1
117            else:
118                self.scan_summary[f"{host}:{port}"][o.plugin.severity] = 1
119
120
121    def export_handler(self, outputs: list[Output], outfile: str):
122        headers = ["HOST", "SEVERITY", "VULN FOUND"]
123        fmt = "github"
124
125        tab = []
126        for host in self.scan_summary.keys():
127            first = True
128            for sev, count in self.scan_summary[host].items():
129                if first:
130                    tab.append([host, sev, count])
131                    first = False
132                else:
133                    tab.append(["", sev, count])
134            tab.append(["-", "-", "-"])
135
136        print(tabulate(tab, headers=headers, tablefmt=fmt))

206    def export_handler(self, outputs: list[Output], outfile: str):
207
208        def complete_log(o: Output):
209            ret = []
210            for i in range(len(o.test_results)):
211                ret.append({

214                    "stderr": o.test_results[i]['stderr'] or ""

216            return ret
217
218        COLUMNS_TO_FUNC = {
219            "Host IP": lambda x: Connection.get_hostname(x.target.connection),
220            "Port": lambda x: Connection.get_port(x.target.connection),
221            "Scheme": lambda x: Connection.get_scheme(x.target.connection),
222            "Vulnerable": lambda x: x.is_vuln_found(),
223            "Any timeout": lambda x: any(x.get_timeouts()),
224            "CVSS score": lambda x: getattr(x.plugin, 'cvss', 0.0),
225            "Vulnerability name": lambda x: getattr(x.plugin, 'name'),
226            "Description": lambda x: getattr(x.plugin, 'description'),
227            "Remediation": lambda x: getattr(x.plugin, 'remediation', ''),
228            "CVE": lambda x: getattr(x.plugin, 'CVE', ''),
229            "CVSS vector": lambda x: getattr(x.plugin, 'cvss_vector', ''),
230            "Complete log": complete_log,
231        }
232        out_dict = []
233
234        # it works only in non multithread environment
235        for output in outputs:
236            tmp_row = {}

245class StarescRawHandler(StarescHandler):
246
247    def import_handler(self, o: Output):
248        pass
249
250    def export_handler(self, outputs: list[Output], outfile: str):
251        from datetime import datetime
252        import os
253        
254        for output in outputs:
255            output_dir = outfile
256            if not output_dir:
257                output_dir = f"staresc_{output.target.hostname}"
258            os.makedirs(output_dir, exist_ok=True)
259
260            base_filename = f"{output.target.hostname}_{datetime.now().strftime('%m-%d_%H.%M.%S')}"
261
262            outstream = "\n\n".join(["$ " + r['stdin'] + "\n" + r['stdout'] for r in output.test_results])
263            errstream = "\n\n".join([r['stderr'] for r in output.test_results])
264            with open(os.path.join(output_dir, base_filename + '.out.log'), 'a+') as f:
265                f.write(outstream)
266            if len(errstream.strip()) > 0:
267                with open(os.path.join(output_dir, base_filename + '.err.log'), 'a+') as f:
268                    f.write(errstream)
